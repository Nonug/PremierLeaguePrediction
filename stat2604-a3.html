AUC helper function

compute.auc <- function(model, test.data)
{
  model.probs = predict(model, test.data[, 1:34], type="prob")
  
  model.roc <- roc(predictor=model.probs$g,
                   response=test.data$V35,
                   levels=rev(levels(test.data$V35)))
  
  model.roc.fp <- 1 - model.roc$specificities
  model.roc.tp <- model.roc$sensitivities
  model.roc.plot <- plot(x = model.roc.fp, y = model.roc.tp, type = "l",
                         xlab = "false positive rate (1 - specificity)",
                         ylab = "true positive rate (sensitivity)",
                         main = "ROC Curve")
  
  model.auc <- model.roc$auc
  
  return(list(
    probs = model.probs,
    roc = model.roc.plot,
    auc = model.auc
  ))
}
